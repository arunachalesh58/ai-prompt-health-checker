}

function getGradeColor(grade) {
  const map = { A: '#22c55e', B: '#84cc16', C: '#f59e0b', D: '#f97316', F: '#ef4444' };
  return map[grade] || '#7a7a8a';
}

function getBarColor(score) {
  if (score >= 70) return '#ef4444';
  if (score >= 40) return '#f59e0b';
  return '#22c55e';
}

function getPiiBarColor(score) {
  if (score >= 70) return '#ef4444';
  if (score >= 30) return '#f59e0b';
  return '#22c55e';
}

function getCostTierWidth(tier) {
  const map = { cheap: 25, moderate: 60, expensive: 90 };
  return map[tier] || 50;
}

function getCostColor(tier) {
  const map = { cheap: '#22c55e', moderate: '#f59e0b', expensive: '#ef4444' };
  return map[tier] || '#7a7a8a';
}

async function analyze() {
  const prompt = document.getElementById('promptInput').value.trim();
  if (!prompt) {
    showError('Please enter a prompt to analyze.');
    return;
  }

  const btn = document.getElementById('analyzeBtn');
  const errorBox = document.getElementById('errorBox');
  const results = document.getElementById('results');

  btn.disabled = true;
  btn.classList.add('loading');
  errorBox.classList.remove('show');
  results.classList.remove('show');

  try {
    const res = await fetch(API_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ prompt })
    });

    const data = await res.json();

    if (!res.ok || data.error) {
      showError(data.error || 'API returned an error. Please try again.');
      return;
    }

    renderResults(data);

  } catch (err) {
    showError('Could not connect to the API. Check your connection and try again.');
  } finally {
    btn.disabled = false;
    btn.classList.remove('loading');
  }
}

function showError(msg) {
  const box = document.getElementById('errorBox');
  box.textContent = msg;
  box.classList.add('show');
}

function renderResults(data) {
  const a = data.analysis;
  const health = a.overall_health;
  const grade = health.grade;
  const gradeColor = getGradeColor(grade);

  const gradeCircle = document.getElementById('gradeCircle');
  gradeCircle.textContent = grade;
  gradeCircle.style.color = gradeColor;
  gradeCircle.style.borderColor = gradeColor;
  gradeCircle.style.boxShadow = `0 0 16px ${gradeColor}40`;

  document.getElementById('overallScore').textContent = health.score + ' / 100';
  document.getElementById('overallScore').style.color = gradeColor;
  document.getElementById('modelTag').textContent = data.model_used || 'amazon.nova-lite-v1:0';

  const pii = a.pii_risk;
  document.getElementById('piiScore').textContent = pii.score + '/100';
  document.getElementById('piiScore').style.color = getPiiBarColor(pii.score);
  document.getElementById('piiBar').style.width = pii.score + '%';
  document.getElementById('piiBar').style.background = getPiiBarColor(pii.score);
  document.getElementById('piiDetail').textContent = 'Level: ' + pii.level;
  const findingsEl = document.getElementById('piiFindings');
  findingsEl.innerHTML = '';
  if (pii.findings && pii.findings.length > 0) {
    pii.findings.forEach(f => {
      const tag = document.createElement('span');
      tag.className = 'finding-tag';
      tag.textContent = f;
      findingsEl.appendChild(tag);
    });
  }

  const drift = a.drift_potential;
  document.getElementById('driftScore').textContent = drift.score + '/100';
  document.getElementById('driftScore').style.color = getBarColor(drift.score);
  document.getElementById('driftBar').style.width = drift.score + '%';
  document.getElementById('driftBar').style.background = getBarColor(drift.score);
  document.getElementById('driftDetail').textContent = drift.reason || ('Level: ' + drift.level);

  const token = a.token_cost;
  const tierW = getCostTierWidth(token.cost_tier);
  document.getElementById('tokenCost').textContent = token.cost_tier;
  document.getElementById('tokenCost').style.color = getCostColor(token.cost_tier);
  document.getElementById('tokenBar').style.width = tierW + '%';
  document.getElementById('tokenBar').style.background = getCostColor(token.cost_tier);
  document.getElementById('tokenDetail').textContent = '~' + token.estimated_tokens + ' tokens · ' + (token.note || '');

  const tone = a.tone;
  document.getElementById('toneLabel').textContent = tone.label;
  document.getElementById('toneLabel').style.color = tone.label === 'aggressive' ? '#ef4444' : tone.label === 'professional' ? '#22c55e' : '#f59e0b';
  document.getElementById('toneBar').style.width = tone.confidence + '%';
  document.getElementById('toneBar').style.background = '#6c63ff';
  document.getElementById('toneDetail').textContent = (tone.confidence || 0) + '% confidence · ' + (tone.note || '');

  document.getElementById('summaryText').textContent = health.summary || '';

  document.getElementById('results').classList.add('show');
  document.getElementById('results').scrollIntoView({ behavior: 'smooth', block: 'nearest' });
}
